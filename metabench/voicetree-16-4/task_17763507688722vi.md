   import base64, sys, tempfile, hashlib
   from pathlib import Path

   EMBEDDED_MODULES = {
       "harness/protocol.py": "<b64>",
       "harness/prompt.py": "<b64>",
       "harness/runner.py": "<b64>",
       "harness/scoring.py": "<b64>",
       "harness/render_nl.py": "<b64>",
       "harness/__init__.py": "",  # if needed
       "verifiers/__init__.py": "<b64>",
       "verifiers/cjs.py": "<b64>",
       ... (all 6 classes) ...
       "generators/__init__.py": "<b64>",
       "generators/cjs.py": "<b64>",
       ... (all 6 classes) ...
   }

   def _ensure_bundle():
       bundle_dir = Path(tempfile.gettempdir()) / "metacog_v1_bundle"
       bundle_dir.mkdir(parents=True, exist_ok=True)
       for filename, encoded in EMBEDDED_MODULES.items():
           target = bundle_dir / filename
           target.parent.mkdir(parents=True, exist_ok=True)
           decoded = base64.b64decode(encoded).decode() if encoded else ""
           if target.exists() and target.read_text() == decoded:
               continue
           target.write_text(decoded)
       p = str(bundle_dir)
       if p not in sys.path:
           sys.path.insert(0, p)

   _ensure_bundle()

   import kaggle_benchmarks as kbench  # kaggle injects this
   import json as _json
   from harness.runner import run_instance
   from harness.scoring import score_portfolio
   from verifiers import CLASS_TO_VERIFIER

   @kbench.task(name="metacog_optim_v1")
   def run(
       llm,
       id: str,
       class_: str,            # column name "class" is a reserved kwarg; use alias — see below
       difficulty: str,
       seed: int,
       instance,               # may arrive as dict or JSON string
       gold_objective: float,
       baseline_objective: float,
       value_cap: float,
       wall_budget_s: int = 1800,
       components=None,
   ) -> float:
       # parse instance if string
       inst = instance if isinstance(instance, dict) else _json.loads(instance)
       if class_ == "portfolio":
           return _run_portfolio(llm, id, seed, components, wall_budget_s)
       row = run_instance(
           llm=llm, instance=inst, cls=class_, difficulty=difficulty, seed=seed,
           gold_objective=gold_objective, baseline_objective=baseline_objective,
           value_cap=value_cap,
       )
       # Attach metadata for analyzer via return-with-metadata? kbench returns float only.
       # Extra fields are handled by logging in runner (acceptable for smoke).
       return float(row["score"])

   def _run_portfolio(llm, id_, seed, components, wall_budget_s):
       # Minimal portfolio dispatch: run solo for each component, aggregate via score_portfolio
       # NOTE: runner's "generic component scoring" partial; keep simple here — S3 scope is wiring, not redesign.
       per_component = []
       for c in (components or []):
           sub_inst = c.get("sub_instance") if isinstance(c, dict) else None
           if sub_inst is None:
               continue
           sub_row = run_instance(
               llm=llm, instance=sub_inst, cls=c["class"], difficulty="medium", seed=seed,
               gold_objective=0.0, baseline_objective=0.0,  # unknown at portfolio level
               value_cap=float(c["value_cap"]),
           )
           per_component.append({
               "value_cap": float(c["value_cap"]),
               "gap_pct": sub_row.get("parsed", {}).get("gap_pct", 100.0),  # fallback
               "feasible": sub_row.get("parsed", {}).get("feasible", False),
           })
       total_wall = sum(pc.get("wall_s", 0.0) for pc in per_component)  # rough
       return float(score_portfolio(per_component, wall_s=total_wall))
